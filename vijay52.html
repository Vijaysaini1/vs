  3. a2-b2=(a+b)(a-b)

  7. 2(a2+b2)=(a+b)2+(a-b)2
  8. (a+b)2+(a-b)2=4ab
  9. a4+b4=(a+b)(a-b)[(a+b)2-2ab]
  10. (a-b)2=(a+b)2-4ab
  11. (a+b)2=(a-b)2+4ab

  14. (a+b-c)2=a2+b2+c2+2ab-2bc-2ca
  15. (a-b-c)2=a2+b2+c2-2ab+2bc-2ca
